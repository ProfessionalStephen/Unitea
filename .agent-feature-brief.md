#### Context: What is ADR-002?

Read: `docs/adr/ADR-002-ralph-automated-loop.md`

**Summary**: RALPH is user-reported issue tracker (already in code). ADR-002 adds auto-categorization and feedback loops so that when a user reports a KPI issue, the system auto-suggests corrections based on patterns.

**Requirement**: Build 5-stage workflow:
- **R** = Reported (user submits issue)
- **A** = Annotating (AI Engineer reviews)
- **L** = Learning (correction mapped to system rules)
- **P** = Patched (fix applied + deployed)
- **H** = Hardened (stress-tested, permanently locked)

```typescript
// Auto-suggest stage based on issue description
export function suggestStage(issueText: string): string {
  const text = issueText.toLowerCase();
  
  // Pattern: user says "KPI shows wrong value" → needs clarification (A)
  if (/wrong|incorrect|bad|fail/i.test(text)) return "A - Annotating";
  
  // Pattern: user says "this metric should come from X" → needs learning (L)
  if (/should|needs to|required|must|has to/i.test(text)) return "L - Learning";
  
  // Pattern: user says "fixed" or "verified" → patched (P)
  if (/fixed|verified|confirm|correct/i.test(text)) return "P - Patched";
  
  // Default: start at reported
  return "R - Reported";
}

// Auto-suggest correction based on KPI name
export function suggestCorrection(kpiName: string, issueText: string): string {
  // Example: if issue is about "M1 invoices needed" showing wrong count
  // Suggestion: "Check if stage name matches exactly: 'M1 Invoice needed' vs 'M1 invoice needed'"
  
  // This is domain-specific; populate with patterns from your domain
  const patterns: Record<string, string> = {
    "M1 invoices needed": "Verify Funding board stage name matches exactly (case-sensitive)",
    "Installs scheduled": "Check if counting 'Installation Scheduled' stage or entire Scheduling/Coordinating board",
    "Net metering pending": "Confirm if counting Net Metering board total or specific stage",
    // Add more patterns
  };
  
  return patterns[kpiName] || "Review KPI source config: board/stage/field mapping";
}

// Auto-suggest AI note (human-readable summary)
export function generateAiNote(stage: string, correction: string): string {
  if (stage === "A - Annotating") {
    return "Issue documented. Awaiting AI Engineer review and clarification.";
  } else if (stage === "L - Learning") {
    return `Suggested fix: "${correction}". Awaiting mapping to resolver rules.`;
  } else if (stage === "P - Patched") {
    return `Applied fix: "${correction}". Awaiting verification and deployment.`;
  } else if (stage === "H - Hardened") {
    return `Permanently locked. Fix verified: "${correction}".`;
  }
  return "";
}
```

#### Step 3: Update RalphFormInline Component

File: `src/App.tsx` (search for RalphFormInline)

**Current behavior**: Form submits, creates RALPH_INIT entry with status="open", stage="R - Reported"

**Enhanced behavior**: 
1. Submit form
2. Call `suggestStage(issueText)` → auto-suggest stage
3. Call `suggestCorrection(kpiName, issueText)` → auto-suggest correction
4. Create entry with suggested stage + correction
5. Call `generateAiNote()` → populate aiNote

```typescript
function handleRalphSubmit(obj) {
  const suggestedStage = suggestStage(obj.issue);
  const suggestedCorrection = suggestCorrection(obj.kpi, obj.issue);
  const aiNote = generateAiNote(suggestedStage, suggestedCorrection);
  
  setRalph((l) => [{
    id: Date.now(),
    ts: new Date().toLocaleString(),
    reporter: obj.reporter,
    issue: obj.issue,
    kpi: obj.kpi,
    status: "open",
    stage: suggestedStage,  // Auto-set based on issue text
    correction: suggestedCorrection,  // Auto-suggest
    aiNote: aiNote  // Auto-generate
  }].concat(l));
  
  setShowRalphForm(false);
  addAudit("RALPH issue auto-triaged", obj.reporter + ": " + obj.issue.slice(0,50), "system");
}
```
